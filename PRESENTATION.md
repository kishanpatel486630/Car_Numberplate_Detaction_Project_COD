# Simplified code flow in app.py

def process_video(video_path):

    # 1. LOAD MODELS
    vehicle_detector = YOLO('yolov8n.pt')      # Detect cars
    plate_detector = YOLO('license_plate.pt')   # Detect plates
    tracker = Sort()                            # Track vehicles

    # 2. OPEN VIDEO
    video = cv2.VideoCapture(video_path)

    # 3. PROCESS EACH FRAME
    for frame in video:

        # Step 1: Detect all vehicles in frame
        vehicles = vehicle_detector.predict(frame)
        #  â†’ Returns: [[x1,y1,x2,y2,conf,class], ...]

        # Step 2: Track vehicles across frames
        tracked = tracker.update(vehicles)
        #  â†’ Returns: [[x1,y1,x2,y2,track_id], ...]

        # Step 3: For each tracked vehicle
        for vehicle in tracked:

            # Crop vehicle region
            vehicle_crop = frame[y1:y2, x1:x2]

            # Detect license plate in crop
            plates = plate_detector.predict(vehicle_crop)

            # Step 4: Read plate text
            for plate in plates:
                plate_crop = vehicle_crop[py1:py2, px1:px2]
                text, confidence = read_license_plate(plate_crop)

                # Save result
                results[frame_num][track_id] = {
                    'car_bbox': [x1,y1,x2,y2],
                    'plate_bbox': [px1,py1,px2,py2],
                    'plate_text': text,
                    'confidence': confidence
                }

    # 4. SAVE RESULTS
    write_csv(results, 'output.csv')
    generate_output_video(video_path, results)
```

# Simplified OCR flow in util.py

def read_license_plate(plate_image):

    # 1. Use EasyOCR to detect text
    detections = ocr_reader.readtext(plate_image)
    #  â†’ Returns: [([box], 'TEXT', confidence), ...]

    for detection in detections:
        text = detection[1].upper()  # Get text, uppercase
        text = text.replace(' ', '') # Remove spaces

        # 2. Validate format (e.g., UK plates: AA00AAA)
        if is_valid_format(text):

            # 3. Fix common OCR mistakes
            #    O looks like 0, I looks like 1, etc.
            text = fix_characters(text)

            return text, confidence

    return None, None
```
